telnet commands for switch:

config t
interface vlan 1
ip address 40.80.3.0 255.0.0.0  (giving each switch an ip)
no shut
exit

line vty 0 15
password cisco
login
end

 *if enable password needed:
config t
enable password cs
exit

-- telnet access:
telnet 40.80.3.0
